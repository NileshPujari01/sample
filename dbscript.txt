CREATE TABLE IF NOT EXISTS users
(
	user_id int not null generated by default as identity,
	CONSTRAINT pk_users PRIMARY KEY (user_id),
	customer_name varchar(60) NOT NULL,
	created_by int NOT NULL DEFAULT 0,
	created_date timestamp with time zone default now(),
	updated_by int NOT NULL DEFAULT 0,
	updated_date timestamp with time zone default now(),
	customer_ic_number int not null,
	UNIQUE(customer_ic_number),
	customer_mobile_no varchar(20) NOT NULL,
	customer_email_id varchar(30) NOT NULL,
	is_privacy_policy_agreed bit,
	is_verified bit,
	app_pin int not null
);
